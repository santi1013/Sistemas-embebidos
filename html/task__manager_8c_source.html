75static void sensor_task(void *pvParameters)
76{
77 mpu_raw_data_t sample;
78 TickType_t last_wake_time = xTaskGetTickCount();
79 uint32_t sample_count = 0;
80
81 ESP_LOGI("SENSOR_TASK", "Tarea MPU6050 iniciada");
82
83 while (1) {
84 if (mpu6050_read_raw(&sample) == ESP_OK) {
85 if (xQueueSend(g_raw_queue, &sample, 0) != pdPASS) {
86 ESP_LOGW("SENSOR_TASK", "g_raw_queue llena, muestra descartada");
87 } else {
88 sample_count++;
89 if ((sample_count % 10) == 0) {
90 ESP_LOGI("SENSOR_TASK",
91 "Enviadas %lu muestras a g_raw_queue | ax=%.3f ay=%.3f az=%.3f",
92 (unsigned long)sample_count,
93 sample.ax, sample.ay, sample.az);
94 }
95 }
96 } else {
97 ESP_LOGE("SENSOR_TASK", "Error leyendo MPU6050");
98 }
99
100 vTaskDelayUntil(&last_wake_time, pdMS_TO_TICKS(SENSOR_SAMPLE_PERIOD_MS));
101 }
102}

117static void format_local_task(void *pvParameters)
118{
119 mpu_raw_data_t sample;
120 log_record_t record;
121
122 float pitch;
123 float roll;
124 uint32_t format_count = 0;
125
126 ESP_LOGI("FORMAT_LOCAL", "Tarea de formateo local iniciada");
127
128 while (1) {
129 if (xQueueReceive(g_raw_queue, &sample, portMAX_DELAY) == pdPASS) {
130
131 roll = atan2f(sample.ay, sample.az) * RAD_TO_DEG;
132 pitch = atan2f(-sample.ax,
133 sqrtf((sample.ay * sample.ay) + (sample.az * sample.az))) * RAD_TO_DEG;
134
135 record.timestamp_ms = sample.timestamp_ms;
136
137 snprintf(record.line, sizeof(record.line),
138 "%lld,%.4f,%.4f,%.4f,%.4f,%.4f,%.4f,%.2f,%.2f,%.2f\n",
139 (long long)sample.timestamp_ms,
140 sample.ax, sample.ay, sample.az,
141 sample.gx, sample.gy, sample.gz,
142 sample.temperature,
143 pitch, roll);
144
145 if (xQueueSend(g_log_queue, &record, portMAX_DELAY) != pdPASS) {
146 ESP_LOGW("FORMAT_LOCAL", "g_log_queue llena");
147 } else {
148 format_count++;
149 if ((format_count % 10) == 0) {
150 ESP_LOGI("FORMAT_LOCAL",
151 "Enviados %lu registros a g_log_queue | pitch=%.2f roll=%.2f",
