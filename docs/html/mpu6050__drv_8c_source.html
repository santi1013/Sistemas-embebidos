170esp_err_t mpu6050_init(void)
171{
172 esp_err_t ret;
173 uint8_t who_am_i = 0;
174
175 ret = mpu6050_i2c_master_init();
176 if (ret != ESP_OK && ret != ESP_ERR_INVALID_STATE) {
177 ESP_LOGE(TAG, "Error inicializando I2C: %s", esp_err_to_name(ret));
178 return ret;
179 }
180
181 ret = mpu6050_read_bytes(MPU6050_REG_WHO_AM_I, &who_am_i, 1);
182 if (ret != ESP_OK) {
183 ESP_LOGE(TAG, "No se pudo leer WHO_AM_I: %s", esp_err_to_name(ret));
184 return ret;
185 }
186
187 ESP_LOGI(TAG, "WHO_AM_I = 0x%02X", who_am_i);
188
189 if (who_am_i != MPU6050_WHO_AM_I_EXPECTED) {
190 ESP_LOGW(TAG, "Valor WHO_AM_I inesperado");
191 }
192
193 /* Despertar el sensor: SLEEP = 0 en PWR_MGMT_1 */
194 ret = mpu6050_write_byte(MPU6050_REG_PWR_MGMT_1, 0x00);
195 if (ret != ESP_OK) {
196 ESP_LOGE(TAG, "No se pudo escribir PWR_MGMT_1: %s", esp_err_to_name(ret));
197 return ret;
198 }
199
200 ESP_LOGI(TAG, "MPU6050 inicializado correctamente");
201 return ESP_OK;
202}

228esp_err_t mpu6050_read_raw(mpu_raw_data_t *out_data)
229{
230 uint8_t raw_buf[14];
231 esp_err_t ret;
232
233 if (out_data == NULL) {
234 return ESP_ERR_INVALID_ARG;
235 }
236
237 ret = mpu6050_read_bytes(MPU6050_REG_ACCEL_XOUT_H, raw_buf, sizeof(raw_buf));
238 if (ret != ESP_OK) {
239 return ret;
240 }
241
242 int16_t ax_raw = (int16_t)((raw_buf[0] << 8) | raw_buf[1]);
243 int16_t ay_raw = (int16_t)((raw_buf[2] << 8) | raw_buf[3]);
244 int16_t az_raw = (int16_t)((raw_buf[4] << 8) | raw_buf[5]);
245 int16_t temp_raw = (int16_t)((raw_buf[6] << 8) | raw_buf[7]);
246 int16_t gx_raw = (int16_t)((raw_buf[8] << 8) | raw_buf[9]);
247 int16_t gy_raw = (int16_t)((raw_buf[10] << 8) | raw_buf[11]);
248 int16_t gz_raw = (int16_t)((raw_buf[12] << 8) | raw_buf[13]);
249
250 /* Escalas por defecto al arrancar:
251 acelerómetro ±2g -> 16384 LSB/g
252 giroscopio ±250 °/s -> 131 LSB/(°/s) */
253 out_data->timestamp_ms = esp_timer_get_time() / 1000;
254 out_data->ax = ax_raw / 16384.0f;
255 out_data->ay = ay_raw / 16384.0f;
256 out_data->az = az_raw / 16384.0f;
257 out_data->gx = gx_raw / 131.0f;
258 out_data->gy = gy_raw / 131.0f;
259 out_data->gz = gz_raw / 131.0f;
260 out_data->temperature = (temp_raw / 340.0f) + 36.53f;
261
262 return ESP_OK;
263}
