102 ret = spi_bus_initialize(host.slot, &bus_cfg, SDSPI_DEFAULT_DMA);
103 if (ret != ESP_OK && ret != ESP_ERR_INVALID_STATE) {
104 ESP_LOGE(TAG, "Error inicializando bus SPI: %s", esp_err_to_name(ret));
105 return ret;
106 }

108 if (ret == ESP_ERR_INVALID_STATE) {
109 ESP_LOGW(TAG, "Bus SPI ya estaba inicializado");
110 } else {
111 ESP_LOGI(TAG, "Bus SPI inicializado correctamente");
112 }
113
114 sdspi_device_config_t slot_config = SDSPI_DEVICE_CONFIG_DEFAULT();
115 slot_config.gpio_cs = SD_SPI_CS;
116 slot_config.host_id = host.slot;
117
118 ret = esp_vfs_fat_sdspi_mount("/sdcard", &host, &slot_config, &mount_config, &g_card);
119 if (ret != ESP_OK) {
120 ESP_LOGE(TAG, "Error montando SD: %s", esp_err_to_name(ret));
121 spi_bus_free(host.slot);
122 return ret;
123 }
124

127 sdmmc_card_print_info(stdout, g_card);

285 if (fprintf(f, "SD_SELF_TEST_OK\n") < 0) {
286 ESP_LOGE(TAG, "Self-test: fallo en fprintf()");
287 fclose(f);

289 }
290
291 if (fflush(f) != 0) {
292 ESP_LOGE(TAG, "Self-test: fallo en fflush()");
293 fclose(f);
294 return ESP_FAIL;
295 }
296
297 fclose(f);
298
299 f = fopen(path, "r");
300 if (f == NULL) {

302 return ESP_FAIL;
303 }
304
305 char line[64];
306 if (fgets(line, sizeof(line), f) != NULL) {
307 line[strcspn(line, "\r\n")] = '\0';
308 ESP_LOGI(TAG, "Self-test OK, primera linea leida: %s", line);
309 } else {
310 ESP_LOGW(TAG, "Self-test: archivo abierto pero sin contenido legible");
311 }
312
313 fclose(f);
314 ESP_LOGI(TAG, "Self-test de microSD completado correctamente");
315 return ESP_OK;
